using Bloomfiy.Models;
using System;
using System.Data.Entity.Migrations;

namespace Bloomfiy.Migrations
{
    internal sealed class Configuration : DbMigrationsConfiguration<ApplicationDbContext>
    {
        public Configuration()
        {
            AutomaticMigrationsEnabled = true;
            AutomaticMigrationDataLossAllowed = true;
        }

        protected override void Seed(ApplicationDbContext context)
        {
            // Seed Categories
            context.Categories.AddOrUpdate(
                c => c.CategoryName,
                new Category { CategoryName = "Bouquets", Description = "Beautiful flower arrangements" },
                new Category { CategoryName = "Single Varieties", Description = "Individual flower types" },
                new Category { CategoryName = "Arrangements", Description = "Designed floral displays" },
                new Category { CategoryName = "Premium Collection", Description = "Luxury floral selections" }
            );

            // Seed Colors
            context.Colors.AddOrUpdate(
                c => c.ColorName,
                new Color { ColorName = "Red", ColorCode = "#FF0000", PriceAdjustment = 0 },
                new Color { ColorName = "White", ColorCode = "#FFFFFF", PriceAdjustment = 0 },
                new Color { ColorName = "Pink", ColorCode = "#FF69B4", PriceAdjustment = 0 },
                new Color { ColorName = "Yellow", ColorCode = "#FFD700", PriceAdjustment = 0 },
                new Color { ColorName = "Purple", ColorCode = "#800080", PriceAdjustment = 0 }
            );

            context.SaveChanges();
        }
    }
}
